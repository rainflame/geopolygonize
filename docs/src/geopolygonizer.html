240    def _clean_tile(
241        self,
242        tile_parameters: TileParameters,
243        tiler_parameters: TilerParameters,
244    ) -> None:
245        step = "clean"
246        data = self._original
247
248        try:
249            tile_path = self._get_path(step, tile_parameters, "npy")
250            if os.path.exists(tile_path):
251                return
252
253            buffer = self._min_blob_size - 1
254            bx0 = int(max(tile_parameters.start_x-buffer, 0))
255            by0 = int(max(tile_parameters.start_y-buffer, 0))
256            bx1 = int(min(
257                tile_parameters.start_x+tile_parameters.width+buffer,
258                tiler_parameters.endx
259            ))
260            by1 = int(min(
261                tile_parameters.start_y+tile_parameters.height+buffer,
262                tiler_parameters.endy
263            ))
264            buffered_tile_raster = data[bx0:bx1, by0:by1]
265
266            blobifier = Blobifier(buffered_tile_raster, self._min_blob_size)
267            cleaned = blobifier.blobify()
268
269            rel_start_x = int(tile_parameters.start_x - bx0)
270            rel_start_y = int(tile_parameters.start_y - by0)
271            rel_end_x = int(min(rel_start_x + tile_parameters.width, bx1))
272            rel_end_y = int(min(rel_start_y + tile_parameters.height, by1))
273            if rel_end_x - rel_start_x <= 0 \
274                    or rel_end_y - rel_start_y <= 0:
275                return
276            tile_raster = cleaned[rel_start_x:rel_end_x, rel_start_y:rel_end_y]
277
278            np.save(tile_path, tile_raster)
279        except Exception as e:
280            self._handle_exception(e, step, tile_parameters)

282    def _clean_stitch(self) -> None:
283        step = "clean"
284        try:
285            file_path = self._get_path(step, None, "npy")
286            if os.path.exists(file_path):
287                return
288
289            raster = np.empty_like(self._original)
290            for filepath in tqdm(
291                glob.glob(self._get_tile_glob(step, "npy")),
292                desc=f"[{step}] Stitching rasters",
293            ):
294                starts = self._get_start_from_file(step, filepath)
295                assert starts is not None, \
296                    f"Could not get start values from {filepath}."
297                start_x, start_y = starts
298
299                tile = np.load(filepath)
300
301                end_x = start_x+tile.shape[0]
302                end_y = start_y+tile.shape[1]
303                assert end_x <= self._endx and end_y <= self._endy, \
304                    "Tile from cleaning step is not within bounds " \
305                    "of input raster size."
306                raster[start_x:end_x, start_y:end_y] = tile
307
308            np.save(file_path, raster)
309        except Exception as e:
310            self._handle_exception(e, step, None)

312    def _polygonize_tile(
313        self,
314        tile_parameters: TileParameters,
315        tiler_parameters: TilerParameters,
316    ) -> None:
317        step = "polygonize"
318        data = self._cleaned
319        assert data is not None
320
321        try:
322            tile_path = self._get_path(step, tile_parameters, "shp")
323            if os.path.exists(tile_path):
324                return
325
326            x0 = int(tile_parameters.start_x)
327            x1 = int(min(
328                tile_parameters.start_x + tile_parameters.width,
329                tiler_parameters.endx
330            ))
331            y0 = int(tile_parameters.start_y)
332            y1 = int(min(
333                tile_parameters.start_y + tile_parameters.height,
334                tiler_parameters.endy
335            ))
336            tile = data[x0:x1, y0:y1]
337
338            shapes_gen = shapes(tile, transform=self._transform)
339            polygons_and_labels = list(zip(*shapes_gen))
340            polygons: List[Polygon] = [
341                shape(s) for s in polygons_and_labels[0]
342            ]
343            labels: List[Any] = [v for v in polygons_and_labels[1]]
344
345            gdf = gpd.GeoDataFrame(geometry=polygons)
346            gdf[self._label_name] = labels
347            # in physical space, x and y are reversed
348            shift_x = tile_parameters.start_y * self._pixel_size
349            shift_y = -(tile_parameters.start_x * self._pixel_size)
350            gdf['geometry'] = gdf['geometry'].apply(
351                lambda geom: translate(geom, xoff=shift_x, yoff=shift_y)
352            )
353            gdf.crs = self._crs
354            gdf.to_file(tile_path)
355        except Exception as e:
356            self._handle_exception(e, step, tile_parameters)

379    def _generate_smoothing_func(self) -> Callable[[LineString], LineString]:
380        def smooth(segment: LineString) -> LineString:
381            coords = chaikins_corner_cutting(
382                segment.coords,
383                self._smoothing_iterations
384            )
385            return LineString(coords)
386
387        return smooth
388
389    def _generate_simplify_func(self) -> Callable[[LineString], LineString]:
390        tolerance = self._pixel_size * self._simplification_pixel_window
391
392        def simplify(segment: LineString) -> LineString:
393            # Simplification will turn rings into what are effectively points.
394            # We cut the ring in half to provide simplification
395            # with non-ring segments instead.
396            if segment.is_ring:
397                assert len(segment.coords) >= 3
398                midpoint_idx = len(segment.coords) // 2
399                segment1 = LineString(segment.coords[:midpoint_idx+1])
400                segment2 = LineString(segment.coords[midpoint_idx:])
401                simplified1 = segment1.simplify(tolerance)
402                simplified2 = segment2.simplify(tolerance)
403                coords =\
404                    list(simplified1.coords)[:-1] + list(simplified2.coords)
405                simplified = LineString(coords)
406            else:
407                simplified = segment.simplify(tolerance)
408            return simplified
409
410        return simplify

412    def _vectorize_tile(
413        self,
414        tile_parameters: TileParameters,
415        tiler_parameters: TilerParameters,
416    ) -> None:
417        step = "vectorize"
418        data = self._polygonized
419        assert data is not None
420
421        try:
422            tile_path = self._get_path(step, tile_parameters, "shp")
423            if os.path.exists(tile_path):
424                return
425
426            x0 = tile_parameters.start_x
427            x1 = min(
428                tile_parameters.start_x + tile_parameters.width,
429                tiler_parameters.endx
430            )
431            y0 = tile_parameters.start_y
432            y1 = min(
433                tile_parameters.start_y + tile_parameters.height,
434                tiler_parameters.endy
435            )
436            mask = Polygon([
437                Point(x0, y0),
438                Point(x0, y1),
439                Point(x1, y1),
440                Point(x1, y0),
441                Point(x0, y0),
442            ])
443            tile = data.clip(mask, keep_geom_type=True)
444
445            raw_geometries = tile.geometry.to_list()
446            raw_labels = tile[self._label_name].to_list()
447            polygons: List[Polygon] = []
448            labels: List[str] = []
449            for geometry, label in list(zip(raw_geometries, raw_labels)):
450                ps: List[Polygon] = []
451                if geometry.geom_type == "Polygon":
452                    ps.append(geometry)
453                elif geometry.geom_type == "MultiPolygon":
454                    ps.extend(list(geometry.geoms))
455                polygons.extend(ps)
456                labels.extend([label] * len(ps))
457            assert len(polygons) == len(labels)
458
459            segmenter = Segmenter(
460                polygons=polygons,
461                labels=labels,
462                pin_border=True,
463            )
464            segmenter.run_per_segment(self._generate_simplify_func())
465            segmenter.run_per_segment(self._generate_smoothing_func())
466            modified_polygons, modified_labels = segmenter.get_result()
467
468            gdf = gpd.GeoDataFrame(geometry=modified_polygons)
469            gdf[self._label_name] = modified_labels
470            gdf.crs = self._crs
471            gdf.to_file(tile_path)
472        except Exception as e:
473            self._handle_exception(e, step, tile_parameters)

507    def geopolygonize(self) -> None:
508        """
509        Process inputs to get output.
510        First, clean the input to take out small blobs of pixels,
511        then vectorize the result to get the final output.
512        """
513
514        step = "clean"
515        tiler_parameters = TilerParameters(
516            step=step,
517            endx=self._endx,
518            endy=self._endy,
519            tile_size=self._tile_size,
520            num_processes=self._workers,
521        )
522        clean_tiler = Tiler(
523            tiler_parameters=tiler_parameters,
524            process_tile=self._clean_tile,
525            stitch_tiles=self._clean_stitch,
526        )
527        clean_tiler.process()
528        clean_file = self._get_path(step, None, "npy")
529        self._cleaned = np.load(clean_file)
530
531        step = "polygonize"
532        tiler_parameters = TilerParameters(
533            step=step,
534            endx=self._endx,
535            endy=self._endy,
536            tile_size=self._tile_size,
537            num_processes=self._workers,
538        )
539        polygonize_tiler = Tiler(
540            tiler_parameters=tiler_parameters,
541            process_tile=self._polygonize_tile,
542            stitch_tiles=self._polygonize_stitch,
543        )
544        polygonize_tiler.process()
545        polygonize_file = self._get_path(step, None, "shp")
546        gdf = gpd.read_file(polygonize_file, encoding="utf-8")
547        self._polygonized = gdf
548
549        assert self._polygonized is not None
550        bounds = self._polygonized.geometry.bounds
551        minx_bound = bounds['minx'].min()
552        maxx_bound = bounds['maxx'].max()
553        miny_bound = bounds['miny'].min()
554        maxy_bound = bounds['maxy'].max()
555
556        step = "vectorize"
557        tiler_parameters = TilerParameters(
558            step=step,
559            startx=minx_bound,
560            starty=miny_bound,
561            endx=maxx_bound,
562            endy=maxy_bound,
563            tile_size=self._tile_size * self._pixel_size,
564            num_processes=self._workers,
565        )
566        vectorize_tiler = Tiler(
567            tiler_parameters=tiler_parameters,
568            process_tile=self._vectorize_tile,
569            stitch_tiles=self._vectorize_stitch,
570        )
571        vectorize_tiler.process()
572        vectorize_file = self._get_path(step, None, "shp")
573
574        self._copy_shapefile(vectorize_file, self._output_file)

241    def _clean_tile(
242        self,
243        tile_parameters: TileParameters,
244        tiler_parameters: TilerParameters,
245    ) -> None:
246        step = "clean"
247        data = self._original
248
249        try:
250            tile_path = self._get_path(step, tile_parameters, "npy")
251            if os.path.exists(tile_path):
252                return
253
254            buffer = self._min_blob_size - 1
255            bx0 = int(max(tile_parameters.start_x-buffer, 0))
256            by0 = int(max(tile_parameters.start_y-buffer, 0))
257            bx1 = int(min(
258                tile_parameters.start_x+tile_parameters.width+buffer,
259                tiler_parameters.endx
260            ))
261            by1 = int(min(
262                tile_parameters.start_y+tile_parameters.height+buffer,
263                tiler_parameters.endy
264            ))
265            buffered_tile_raster = data[bx0:bx1, by0:by1]
266
267            blobifier = Blobifier(buffered_tile_raster, self._min_blob_size)
268            cleaned = blobifier.blobify()
269
270            rel_start_x = int(tile_parameters.start_x - bx0)
271            rel_start_y = int(tile_parameters.start_y - by0)
272            rel_end_x = int(min(rel_start_x + tile_parameters.width, bx1))
273            rel_end_y = int(min(rel_start_y + tile_parameters.height, by1))
274            if rel_end_x - rel_start_x <= 0 \
275                    or rel_end_y - rel_start_y <= 0:
276                return
277            tile_raster = cleaned[rel_start_x:rel_end_x, rel_start_y:rel_end_y]
278
279            np.save(tile_path, tile_raster)
280        except Exception as e:
281            self._handle_exception(e, step, tile_parameters)

305                    "Tile from cleaning step is not within bounds " \
306                    "of input raster size."

318        step = "polygonize"

320        assert data is not None
321
322        try:
323            tile_path = self._get_path(step, tile_parameters, "shp")

380    def _generate_smoothing_func(self) -> Callable[[LineString], LineString]:
381        def smooth(segment: LineString) -> LineString:
382            coords = chaikins_corner_cutting(
383                segment.coords,
384                self._smoothing_iterations
385            )
386            return LineString(coords)
387
388        return smooth

390    def _generate_simplify_func(self) -> Callable[[LineString], LineString]:
391        tolerance = self._pixel_size * self._simplification_pixel_window
392
393        def simplify(segment: LineString) -> LineString:
394            # Simplification will turn rings into what are effectively points.
395            # We cut the ring in half to provide simplification
396            # with non-ring segments instead.
397            if segment.is_ring:
398                assert len(segment.coords) >= 3
399                midpoint_idx = len(segment.coords) // 2
400                segment1 = LineString(segment.coords[:midpoint_idx+1])
401                segment2 = LineString(segment.coords[midpoint_idx:])
402                simplified1 = segment1.simplify(tolerance)
403                simplified2 = segment2.simplify(tolerance)
404                coords =\
405                    list(simplified1.coords)[:-1] + list(simplified2.coords)
406                simplified = LineString(coords)
407            else:
408                simplified = segment.simplify(tolerance)
409            return simplified
410
411        return simplify

414        self,
415        tile_parameters: TileParameters,
416        tiler_parameters: TilerParameters,
417    ) -> None:
418        step = "vectorize"
419        data = self._polygonized
420        assert data is not None

422        try:
423            tile_path = self._get_path(step, tile_parameters, "shp")
424            if os.path.exists(tile_path):
425                return
426
427            x0 = tile_parameters.start_x
428            x1 = min(
429                tile_parameters.start_x + tile_parameters.width,
430                tiler_parameters.endx
431            )
432            y0 = tile_parameters.start_y
433            y1 = min(
434                tile_parameters.start_y + tile_parameters.height,

438                Point(x0, y0),
