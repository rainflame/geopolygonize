 95            "--pixel-size",

 99
100        self._check_is_non_negative(
101            "--simplification-pixel-window",
102            params.simplification_pixel_window
103        )

105
106        self._check_is_non_negative(
107            "--smoothing-iterations",
108            params.smoothing_iterations
109        )
110        self._smoothing_iterations = params.smoothing_iterations
111
112        self._check_is_positive(
113            "--tile-size",
114            params.tile_size

116        self._tile_size = params.tile_size

118        self._check_is_non_negative(
119            "--workers",

123            if params.workers == 0 else params.workers
124

127        with rasterio.open(params.input_file) as src:
128            self._data: np.ndarray = src.read(1)
129            self._meta: Dict[str, Any] = src.meta
130            self._crs: CRS = self._meta['crs']
131            self._transform: Affine = src.transform
132
133            if params.pixel_size == 0:
134                # assume pixel is square
135                assert abs(src.res[0] - src.res[1]) < _EPSILON
136                pixel_size = abs(src.res[0])
137                if pixel_size == 0:
138                    raise RuntimeError(
139                        "Cannot infer pixel size from input file. "
140                        "Please input it manually using `--pixel-size`."
141                    )

143
144            self._endx: int = self._data.shape[0]
145            self._endy: int = self._data.shape[1]
146
147        fd, filepath = tempfile.mkstemp()
148        print(f"Tile errors logged in {filepath}")

167    def _clean(self, tile: np.ndarray) -> np.ndarray:
168        blobifier = Blobifier(tile, self._min_blob_size)

172    def _generate_smoothing_func(self) -> Callable[[LineString], LineString]:
173        def smooth(segment: LineString) -> LineString:
174            coords = chaikins_corner_cutting(
175                segment.coords,
176                self._smoothing_iterations
177            )
178            return LineString(coords)
179
180        return smooth
181
182    def _generate_simplify_func(self) -> Callable[[LineString], LineString]:
183        tolerance = self._pixel_size * self._simplification_pixel_window
184
185        def simplify(segment: LineString) -> LineString:
186            # Simplification will turn rings into what are effectively points.
187            # We cut the ring in half to provide simplification
188            # with non-ring segments instead.
189            if segment.is_ring:
190                assert len(segment.coords) >= 3
191                midpoint_idx = len(segment.coords) // 2
192                segment1 = LineString(segment.coords[:midpoint_idx+1])
193                segment2 = LineString(segment.coords[midpoint_idx:])
194                simplified1 = segment1.simplify(tolerance)
195                simplified2 = segment2.simplify(tolerance)
196                coords =\
197                    list(simplified1.coords)[:-1] + list(simplified2.coords)
198                simplified = LineString(coords)
199            else:
200                simplified = segment.simplify(tolerance)
201            return simplified
202
203        return simplify

205    def _vectorize(self, tile: np.ndarray) -> gpd.GeoDataFrame:
206        shapes_gen = shapes(tile, transform=self._transform)
207        polygons_and_labels = list(zip(*shapes_gen))
208        polygons: List[Polygon] = [shape(s) for s in polygons_and_labels[0]]
209        labels: List[Any] = [v for v in polygons_and_labels[1]]
210
211        segmenter = Segmenter(
212            polygons=polygons,
213            labels=labels,
214            pin_border=True,
215        )
216        segmenter.run_per_segment(self._generate_simplify_func())
217        segmenter.run_per_segment(self._generate_smoothing_func())
218        modified_polygons, modified_labels = segmenter.get_result()
219
220        gdf = gpd.GeoDataFrame(geometry=modified_polygons)
221        gdf[self._label_name] = modified_labels
222        return gdf

224    def _process_tile(
225        self,
226        tile_parameters: TileParameters,
227        tiler_parameters: TilerParameters,
228    ) -> None:
229        try:
230            buffer = self._min_blob_size - 1
231
232            bx0 = max(tile_parameters.start_x-buffer, 0)
233            by0 = max(tile_parameters.start_y-buffer, 0)
234            bx1 = min(
235                tile_parameters.start_x+tile_parameters.width+buffer,
236                tiler_parameters.endx
237            )
238            by1 = min(
239                tile_parameters.start_y+tile_parameters.height+buffer,
240                tiler_parameters.endy
241            )
242
243            if bx1 - bx0 <= 2 * buffer or by1 - by0 <= 2 * buffer:
244                empty = gpd.GeoDataFrame({"geometry": []})
245                empty = empty.set_geometry("geometry")
246                return empty
247
248            tile_raster = self._data[bx0:bx1, by0:by1]
249            cleaned = self._clean(tile_raster)
250
251            rel_start_x = tile_parameters.start_x - bx0
252            rel_start_y = tile_parameters.start_y - by0
253            rel_end_x = min(rel_start_x + tile_parameters.width, bx1)
254            rel_end_y = min(rel_start_y + tile_parameters.height, by1)
255
256            unbuffered = cleaned[rel_start_x:rel_end_x, rel_start_y:rel_end_y]
257            gdf = self._vectorize(unbuffered)
258
259            # in physical space, x and y are reversed
260            shift_x = tile_parameters.start_y * self._pixel_size
261            shift_y = -(tile_parameters.start_x * self._pixel_size)
262            gdf['geometry'] = gdf['geometry'].apply(
263                lambda geom: translate(geom, xoff=shift_x, yoff=shift_y)
264            )
265            gdf.crs = self._crs
266
267            gdf.to_file(os.path.join(
268                self._temp_dir,
269                "tile-"
270                f"{tile_parameters.start_x}-{tile_parameters.start_y}.shp",
271            ))
272        except Exception as e:
273            message = f"[{os.getpid()}] Exception at " \
274                f"({tile_parameters.start_x}, {tile_parameters.start_y}): " \
275                f"{e}\n"
276            with os.fdopen(self._log_fd, 'w') as tmp:
277                tmp.write(message)
278
279    def _stitch_tiles(self) -> gpd.GeoDataFrame:
280        all_gdfs = []
281
282        for filepath in tqdm(
283            glob.glob(os.path.join(self._temp_dir, "*.shp")),
284            desc="Stitching tiles",
285        ):
286            gdf = gpd.read_file(filepath)
287            all_gdfs.append(gdf)
288
289        output_gdf = pd.concat(all_gdfs)
290        output_gdf = unify_by_label(output_gdf, self._label_name)
291        return output_gdf
292
293    def geopolygonize(self) -> None:
294        """
295        Process inputs to get output.
296        """
297
298        tiler_parameters = TilerParameters(
299            endx=self._endx,
300            endy=self._endy,
301            tile_size=self._tile_size,
302            num_processes=self._workers,
303        )
304        rz = Tiler(
305            tiler_parameters=tiler_parameters,
306            process_tile=self._process_tile,
307            stitch_tiles=self._stitch_tiles,
308        )
309        gdf = rz.process()
310        gdf.to_file(self._output_file)
311
312        shutil.rmtree(self._temp_dir)
313        os.close(self._log_fd)
